7. 二叉树最近公共祖先问题（无父节点）。
	
	TreeNode *getLCA(TreeNode *root, TreeNode *p, TreeNode *q)
	 {
		if (!root) return nullptr;
		else if (root == p || root == q) return root;
		
		TreeNode *L = getLCA(root->left, p, q);
		TreeNOde *R = getLCA(root->right, p, q);
		if (L && R) return root;
		else return L ? L : R;
	}
